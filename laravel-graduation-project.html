app/Http/Controllers/Auth/RegisterController.php
class RegisterController extends Controller
{
    public function showRegistrationForm(): View
    {
        return view('auth.register');
    }

    public function register(RegisterRequest $request): RedirectResponse
    {
        $user = User::create([
            'name'     => $request->name,
            'email'    => $request->email,
            'password' => bcrypt($request->password),
            'role'     => $request->role ?? 'customer',
        ]);

        $user->assignRole($user->role);  // Spatie Permission

        event(new Registered($user));  // fires email verification

        auth()->login($user);

        return redirect()->route('dashboard')
                    ->with('success', 'Account created successfully!');
    }
}
app/Http/Middleware/RoleMiddleware.php
class RoleMiddleware
{
    public function handle(Request $request, Closure $next, string ...$roles): Response
    {
        if (!auth()->check()) {
            return redirect()->route('login');
        }

        foreach ($roles as $role) {
            if (auth()->user()->hasRole($role)) {
                return $next($request);
            }
        }

        return abort(403, 'Unauthorized action.');
    }
}

Policies

ProductPolicy

  • viewAny: everyone
  • view: everyone
  • create: vendor, admin
  • update: owner vendor, admin
  • delete: owner vendor, admin

OrderPolicy

  • viewAny: admin only
  • view: owner customer
  • update: vendor (status), admin
  • cancel: customer (pending only)

ReviewPolicy

  • create: customer (purchased)
  • update: own review only
  • delete: admin, own review
  • approve: admin only

database/seeders/RolesAndPermissionsSeeder.php
class RolesAndPermissionsSeeder extends Seeder
{
    public function run(): void
    {
        app()[PermissionRegistrar::class]->forgetCachedPermissions();

        $permissions = [
            'products.create', 'products.edit', 'products.delete', 'products.view',
            'orders.view', 'orders.update', 'orders.cancel',
            'users.manage', 'vendors.approve', 'categories.manage',
            'coupons.manage', 'reviews.moderate', 'reports.view',
        ];

        foreach ($permissions as $perm) {
            Permission::create(['name' => $perm]);
        }

        // Admin gets all permissions
        Role::create(['name' => 'admin'])->givePermissionTo(Permission::all());

        // Vendor gets product + order permissions
        Role::create(['name' => 'vendor'])->givePermissionTo([
            'products.create', 'products.edit', 'products.delete',
            'orders.view', 'orders.update',
        ]);

        // Customer gets basic permissions
        Role::create(['name' => 'customer'])->givePermissionTo([
            'orders.view', 'orders.cancel',
        ]);
    }
}

app/Http/Controllers/Vendor/ProductController.php
class ProductController extends Controller
{
    public function __construct(private readonly ProductService $productService)
    {
        $this->middleware('auth');
        $this->middleware('role:vendor,admin');
    }

    public function index(): View
    {
        $products = auth()->user()->products()
            ->with(['category', 'variants'])
            ->latest()
            ->paginate(15);

        return view('vendor.products.index', compact('products'));
    }

    public function store(StoreProductRequest $request): RedirectResponse
    {
        $product = $this->productService->create(
            auth()->user(),
            $request->validated(),
            $request->file('images')
        );

        return redirect()->route('vendor.products.show', $product)
                    ->with('success', 'Product created! Pending admin review.');
    }

    public function update(UpdateProductRequest $request, Product $product): RedirectResponse
    {
        $this->authorize('update', $product);  // ProductPolicy check

        $this->productService->update($product, $request->validated());

        return back()->with('success', 'Product updated successfully.');
    }

    public function destroy(Product $product): RedirectResponse
    {
        $this->authorize('delete', $product);
        $product->delete();  // soft delete

        return redirect()->route('vendor.products.index')
                    ->with('success', 'Product deleted.');
    }
}
app/Http/Controllers/Customer/CheckoutController.php
class CheckoutController extends Controller
{
    public function __construct(
        private readonly CartService     $cart,
        private readonly OrderService    $orders,
        private readonly PaymentService  $payments,
    ) {}

    public function show(): View
    {
        $cartItems  = $this->cart->getItems(auth()->user());
        $addresses  = auth()->user()->addresses;
        $totals     = $this->cart->calculateTotals($cartItems);

        return view('checkout.index', compact('cartItems', 'addresses', 'totals'));
    }

    public function placeOrder(PlaceOrderRequest $request): RedirectResponse
    {
        $order = $this->orders->createFromCart(
            auth()->user(),
            $request->validated()
        );

        $paymentIntent = $this->payments->createStripeIntent($order);

        return redirect()->route('checkout.payment', [
            'order'  => $order->id,
            'intent' => $paymentIntent->client_secret
        ]);
    }
}

Key Models & Relationships

app/Models/User.php
class User extends Authenticatable
{
    use HasRoles, HasFactory, Notifiable, SoftDeletes;

    // Relationships
    public function vendorProfile(): HasOne
    {
        return $this->hasOne(VendorProfile::class);
    }

    public function products(): HasMany
    {
        return $this->hasMany(Product::class, 'vendor_id');
    }

    public function orders(): HasMany
    {
        return $this->hasMany(Order::class);
    }

    public function addresses(): HasMany
    {
        return $this->hasMany(Address::class);
    }

    public function cartItems(): HasMany
    {
        return $this->hasMany(CartItem::class);
    }

    public function wishlists(): BelongsToMany
    {
        return $this->belongsToMany(Product::class, 'wishlists')->withTimestamps();
    }

    // Scopes
    public function scopeVendors($query) { return $query->role('vendor'); }
    public function scopeCustomers($query) { return $query->role('customer'); }
}

Grading Tips

Code Quality

Use Form Requests for validation, Service classes for business logic, Policies for authorization. Avoid fat controllers.

Database Design

Proper foreign keys, indexes on searched columns, use soft deletes for products/users. Normalize to 3NF at minimum.

Security

CSRF on all forms, policy checks on every action, SQL injection safe (use Eloquent), XSS protection via Blade escaping.

Testing

Write at least 20 Pest tests covering auth, product CRUD, order flow, and permission checks. Run with php artisan test.
